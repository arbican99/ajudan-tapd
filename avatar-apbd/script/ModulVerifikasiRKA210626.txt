import React, { useState, useEffect } from 'react';
import { FileSpreadsheet, Layers, Save, Download, ChevronDown, ChevronRight } from 'lucide-react';
import { supabase } from '../supabaseClient';
import * as XLSX from 'xlsx';

export default function ModulVerifikasiRka() {
  const [listTahapan, setListTahapan] = useState([]);
  const [listSkpd, setListSkpd] = useState([]);
  const [rules, setRules] = useState([]);
  const [masterTematik, setMasterTematik] = useState([]); 
  const [loading, setLoading] = useState(false);
  const [saving, setSaving] = useState(false);
  
  const [selectedTahapan, setSelectedTahapan] = useState('');
  const [selectedSkpd, setSelectedSkpd] = useState(''); 
  const [namaFile, setNamaFile] = useState('');
  
  // State manajemen struktur data
  const [subKegiatanTree, setSubKegiatanTree] = useState({});
  const [subUnitGroupData, setSubUnitGroupData] = useState({});
  const [tematikReport, setTematikReport] = useState({}); 
  const [reportGenerated, setReportGenerated] = useState(false);

  // State untuk melacak baris sub-kegiatan yang sedang di-expand
  const [expandedSubKeg, setExpandedSubKeg] = useState({});

  useEffect(() => {
    fetchMasterData();
  }, []);

  const fetchMasterData = async () => {
    setLoading(true);
    try {
      const { data: statusData } = await supabase.from('tblstatus').select('*').order('kd_status');
      setListTahapan(statusData || []);

      const { data: skpdData } = await supabase.from('tblskpd').select('*');
      
      // Mengeliminasi duplikasi entitas agar tampil satu saja di combo box
      const uniqueSkpdMap = {};
      (skpdData || []).forEach(item => {
        let fullSkpdCode = String(item.kd_skpd || '').trim();
        let namaResmi = String(item.nm_skpd || 'Unit Perangkat Daerah').trim();

        if (namaResmi.toUpperCase().includes('DINAS KESEHATAN')) {
          if (!uniqueSkpdMap[fullSkpdCode]) {
            uniqueSkpdMap[fullSkpdCode] = {
              kd_skpd_utama: fullSkpdCode, 
              display_nama: 'DINAS KESEHATAN (KOMPILASI)',
              isDinkes: true
            };
          }
        } else {
          uniqueSkpdMap[fullSkpdCode] = {
            kd_skpd_utama: fullSkpdCode,
            display_nama: namaResmi,
            isDinkes: false
          };
        }
      });

      // PERBAIKAN URUTAN COMBO BOX: Diurutkan secara alfabetis & numerik murni berdasarkan KODE SKPD UTAMA
      const uniqueSkpdList = Object.values(uniqueSkpdMap).sort((a, b) => {
        return a.kd_skpd_utama.localeCompare(b.kd_skpd_utama, undefined, { numeric: true, sensitivity: 'base' });
      });
      
      setListSkpd(uniqueSkpdList);

      const { data: ruleData } = await supabase.from('tbl_rule_validasi').select('*');
      setRules(ruleData || []);

      const { data: mdsData } = await supabase.from('tblmds').select('katagori, kode, kdsubgiat, nmsubgiat');
      setMasterTematik(mdsData || []);

    } catch (err) {
      console.error('Gagal sinkronisasi data master:', err);
    } finally {
      setLoading(false);
    }
  };

  const toggleTreeRow = (subKegCode) => {
    setExpandedSubKeg(prev => ({
      ...prev,
      [subKegCode]: !prev[subKegCode]
    }));
  };

  const handleUploadExcel = (e) => {
    const file = e.target.files[0];
    if (!file) return;
    
    if (!selectedTahapan || !selectedSkpd) {
      alert('PERINGATAN: Silakan tentukan Status Tahapan APBD dan Perangkat Daerah terlebih dahulu sebelum mengunggah berkas Excel!');
      e.target.value = null; 
      setNamaFile('');
      setReportGenerated(false);
      return;
    }

    setNamaFile(file.name);
    setLoading(true);
    
    const reader = new FileReader();
    reader.onload = (evt) => {
      try {
        const bstr = evt.target.result;
        const wb = XLSX.read(bstr, { type: 'binary' });
        const wsname = wb.SheetNames[0];
        const ws = wb.Sheets[wsname];
        
        const rawData = XLSX.utils.sheet_to_json(ws);
        if (!rawData || rawData.length === 0) {
          alert('Format Error: File Excel terdeteksi kosong.');
          setLoading(false);
          return;
        }

        prosesSimulasiAuditTree(rawData);

      } catch (err) {
        alert('Gagal membaca struktur file Excel.');
        console.error('Error Reader Excel:', err);
      } finally {
        setLoading(false);
        e.target.value = null;
      }
    };
    reader.readAsBinaryString(file);
  };

  const prosesSimulasiAuditTree = (dataRows) => {
    const targetSkpdObj = listSkpd.find(s => s.kd_skpd_utama === selectedSkpd);
    const isTargetDinkes = targetSkpdObj?.isDinkes;

    let dataSkpdValid = true;
    let sampelKodeExcel = "";

    for (let i = 0; i < dataRows.length; i++) {
      const rCode = String(dataRows[i]['Kode SKPD'] || dataRows[i]['KODE SKPD'] || dataRows[i]['kode_skpd'] || '').trim();
      if (rCode) {
        sampelKodeExcel = rCode;
        if (isTargetDinkes) {
          const prefixTarget = selectedSkpd.split('.')[0] + '.' + (selectedSkpd.split('.')[1] || '');
          if (!rCode.startsWith(prefixTarget) && !rCode.startsWith(selectedSkpd.substring(0, 4))) { 
            dataSkpdValid = false;
            break;
          }
        } else {
          if (rCode !== selectedSkpd && !rCode.startsWith(selectedSkpd)) {
            dataSkpdValid = false;
            break;
          }
        }
        break; 
      }
    }

    if (!dataSkpdValid) {
      alert(`❌ PENOLAKAN SYSTEM: File Excel tidak sesuai dengan SKPD yang dipilih!\n\nSKPD Terpilih: ${targetSkpdObj?.display_nama} (${selectedSkpd})\nKode Pada Excel: ${sampelKodeExcel}`);
      setReportGenerated(false);
      setNamaFile('');
      return;
    }

    const dapatkanPesanAturan = (idAturan, pesanDefault) => {
      const r = rules.find(item => item.id_aturan === idAturan);
      return r ? r.error_message : pesanDefault;
    };

    const treeGroup = {};
    const tematikGroup = {}; 
    const unitHierarkiGroup = {}; 
    let totalAnomaliGlobal = 0;

    dataRows.forEach((row) => {
      const rawSubUnitExcel = String(row['Kode SKPD'] || row['KODE SKPD'] || row['kode_skpd'] || '').trim();
      const skpdIndukExcel = rawSubUnitExcel;
      const namaSubUnitExcel = String(row['Nama SKPD'] || row['NAMA SKPD'] || row['Nama Unit'] || row['nama_skpd'] || 'Unit Kerja').trim().toUpperCase();
      const namaBidangUrusanExcel = String(row['Bidang Urusan'] || row['BIDANG URUSAN'] || 'URUSAN PEMERINTAHAN').trim().toUpperCase();

      const subKeg = String(row['Kode Sub Kegiatan'] || row['KODE SUB KEGIATAN'] || row['kode_sub_kegiatan'] || '').trim();
      const namaSub = String(row['Nama Sub Kegiatan'] || row['NAMA SUB KEGIATAN'] || row['nama_sub_kegiatan'] || '').trim();
      const namaSubLower = namaSub.toLowerCase();
      
      const rek = String(row['Kode Rekening'] || row['KODE REKENING'] || row['kode_rekening'] || '').trim();
      const namaRek = String(row['Nama Rekening'] || row['NAMA REKENING'] || row['nama_rekening'] || '').trim();
      const namaRekLower = namaRek.toLowerCase();
      const nominal = parseFloat(row['Anggaran'] || row['ANGGARAN'] || row['jumlah'] || row['Jumlah'] || row['nilai'] || 0);

      if (!subKeg || !rawSubUnitExcel) return;

      // VARIABEL KONDISI VALIDASI REGULASI
      const apakahSubKegiatanFisik = subKeg.startsWith('1.03.02') || 
                                     namaSubLower.includes('pembangunan') || 
                                     namaSubLower.includes('rehabilitasi') || 
                                     namaSubLower.includes('rekonstruksi');
                                     
      const apakahKonsultanFisik = rek.startsWith('5.1.02.02.08') || 
                                   rek.startsWith('5.1.02.02.008') || 
                                   namaRekLower.includes('jasa konsultansi berorientasi layanan') ||
                                   namaRekLower.includes('konsultan perencana') ||
                                   namaRekLower.includes('konsultan pengawas');

      // Tabel 2: Pemetaan Tematik
      const angkaMurniSubKegExcel = subKeg.replace(/[^0-9]/g, '');
      if (angkaMurniSubKegExcel.length > 0) {
        const matchTematik = masterTematik.find(t => {
          const angkaMurniMaster1 = String(t.kdsubgiat || '').replace(/[^0-9]/g, '');
          const angkaMurniMaster2 = String(t.kode || '').replace(/[^0-9]/g, '');
          return (angkaMurniMaster1 === angkaMurniSubKegExcel) || (angkaMurniMaster2 === angkaMurniSubKegExcel);
        });

        if (matchTematik && matchTematik.katagori) {
          const namaKategoriMurni = String(matchTematik.katagori).toUpperCase().trim();
          if (!tematikGroup[namaKategoriMurni]) {
            tematikGroup[namaKategoriMurni] = { kategori: namaKategoriMurni, bel_utama: 0, bel_pendukung: 0, jumlah: 0 };
          }
          let jenisKomponenTematik = 'pendukung';
          if (rek.startsWith('5.2')) {
            jenisKomponenTematik = 'utama';
          } else if (rek.startsWith('5.1') && apakahSubKegiatanFisik && apakahKonsultanFisik) {
            jenisKomponenTematik = 'utama';
          }

          if (jenisKomponenTematik === 'utama') tematikGroup[namaKategoriMurni].bel_utama += nominal;
          else tematikGroup[namaKategoriMurni].bel_pendukung += nominal;
          tematikGroup[namaKategoriMurni].jumlah += nominal;
        }
      }

      const treeUniqueKey = `${rawSubUnitExcel}_${subKeg}`;

      if (!treeGroup[treeUniqueKey]) {
        treeGroup[treeUniqueKey] = {
          kd_skpd_utama: selectedSkpd,
          kd_subunit: rawSubUnitExcel,
          nama_subunit: namaSubUnitExcel,
          nama_bidang_urusan: namaBidangUrusanExcel,
          kd_sub_kegiatan: subKeg,
          nama_sub_kegiatan: namaSub,
          total_pagu_sub: 0,
          pagu_utama: 0,      
          pagu_penunjang: 0,   
          rekeningRows: {}, 
          subErrors: [],
          b_pegawai: 0, b_barjas: 0, b_hibah: 0, b_modal_alat: 0, b_modal_gedung: 0, b_modal_jalan: 0,
          b_modal_aset_tetap: 0, b_modal_aset_lain: 0, b_btt: 0, b_bagi_hasil: 0, b_bantuan_keu: 0
        };
      }

      treeGroup[treeUniqueKey].total_pagu_sub += nominal;
      
      let isUtama = false;
      if (rek.startsWith('5.2')) {
        isUtama = true;
        treeGroup[treeUniqueKey].pagu_utama += nominal;
      } else {
        if (rek.startsWith('5.1') && apakahSubKegiatanFisik && apakahKonsultanFisik) {
          isUtama = true;
          treeGroup[treeUniqueKey].pagu_utama += nominal;
        } else {
          treeGroup[treeUniqueKey].pagu_penunjang += nominal;
        }
      }

      if (rek) {
        if (!treeGroup[treeUniqueKey].rekeningRows[rek]) {
          treeGroup[treeUniqueKey].rekeningRows[rek] = {
            kd_rekening: rek,
            nama_rekening: namaRek || 'Rincian Objek Belanja',
            pagu_utama_rek: 0,
            pagu_penunjang_rek: 0,
            total_rek: 0,
            errors: []
          };
        }
        treeGroup[treeUniqueKey].rekeningRows[rek].total_rek += nominal;
        if (isUtama) treeGroup[treeUniqueKey].rekeningRows[rek].pagu_utama_rek += nominal;
        else treeGroup[treeUniqueKey].rekeningRows[rek].pagu_penunjang_rek += nominal;
      }

      // Tabel 3: Klasifikasi Jenis Belanja
      if (rek.startsWith('5.1.01')) treeGroup[treeUniqueKey].b_pegawai += nominal;
      else if (rek.startsWith('5.1.02') && !rek.startsWith('5.1.02.03')) treeGroup[treeUniqueKey].b_barjas += nominal;
      else if (rek.startsWith('5.1.02.03')) treeGroup[treeUniqueKey].b_hibah += nominal;
      else if (rek.startsWith('5.2.02')) treeGroup[treeUniqueKey].b_modal_alat += nominal;
      else if (rek.startsWith('5.2.03')) treeGroup[treeUniqueKey].b_modal_gedung += nominal;
      else if (rek.startsWith('5.2.04')) treeGroup[treeUniqueKey].b_modal_jalan += nominal;
      else if (rek.startsWith('5.2.05')) treeGroup[treeUniqueKey].b_modal_aset_tetap += nominal;
      else if (rek.startsWith('5.2.06')) treeGroup[treeUniqueKey].b_modal_aset_lain += nominal;
      else if (rek.startsWith('5.3')) treeGroup[treeUniqueKey].b_btt += nominal;
      else if (rek.startsWith('5.4.01')) treeGroup[treeUniqueKey].b_bagi_hasil += nominal;
      else if (rek.startsWith('5.4.02')) treeGroup[treeUniqueKey].b_bantuan_keu += nominal;

      // EVALUASI LOG REGULASI
      let barisErrors = [];
      
      // R-01: Validasi Kesesuaian Kewenangan Urusan
      if (!subKeg.endsWith('.01') && subKeg.length >= 4 && skpdIndukExcel.length >= 4) {
        if (skpdIndukExcel.substring(0, 4) !== subKeg.substring(0, 4)) {
          barisErrors.push(dapatkanPesanAturan('R-01', 'Gagal Validasi Kewenangan! SKPD dilarang menganggarkan Sub-Kegiatan di luar bidang urusan wajibnya.'));
        }
      }

      // R-02: Salah Akun Konsultan Perencana/Pengawas Fisik 
      if (apakahSubKegiatanFisik && apakahKonsultanFisik && nominal > 0) {
        barisErrors.push(`[${rek}] ` + dapatkanPesanAturan('R-02', 'Pencampuran Akun! Untuk sub-kegiatan konstruksi fisik, Jasa Konsultan harus menggunakan akun Belanja Modal (awalan 5.2) agar nilainya dikapitalisasi ke aset tetap.'));
      }

      // R-04: SKPD Administratif melompati Batas Belanja Jalan/Irigasi/Jaringan
      const dinasTeknis = ['1.03', '1.04', '2.10'];
      if (!dinasTeknis.some(pfx => skpdIndukExcel.startsWith(pfx)) && rek.startsWith('5.2.04') && nominal > 0) {
        barisErrors.push(dapatkanPesanAturan('R-04', 'Pelanggaran Aturan! SKPD Administratif dilarang menganggarkan Belanja Modal Jalan, Irigasi, dan Jaringan.'));
      }

      // R-05: Pagu Modal Terlalu Besar di Sub-Kegiatan Penunjang Rutin Kantor
      const subKegParts = subKeg.split('.');
      const akhiranSubKeg = subKegParts[subKegParts.length - 1];
      const apakahSubKegiatanRutinAparatur = subKeg.includes('.01.') || akhiranSubKeg === '0001' || akhiranSubKeg === '0002' || namaSubLower.includes('penunjang urusan') || namaSubLower.includes('administrasi perangkat daerah');
      if (apakahSubKegiatanRutinAparatur && !namaSubLower.includes('pembangunan') && !namaSubLower.includes('pengadaan') && (rek.startsWith('5.2.03') || rek.startsWith('5.2.04')) && nominal > 50000000) {
        barisErrors.push(dapatkanPesanAturan('R-05', 'Salah Penempatan! Sub-kegiatan penunjang administrasi kantor rutin hanya untuk operasional.'));
      }

      if (barisErrors.length > 0) {
        totalAnomaliGlobal += barisErrors.length;
        barisErrors.forEach(errTxt => {
          if (!treeGroup[treeUniqueKey].rekeningRows[rek].errors.includes(errTxt)) {
            treeGroup[treeUniqueKey].rekeningRows[rek].errors.push(errTxt);
          }
        });
      }
    });

    // Penggabungan kompilasi error ke subErrors tingkat sub-unit
    Object.values(treeGroup).forEach((subNode) => {
      // R-03: Perbandingan Rasio Belanja Utama vs Pendukung
      if (subNode.pagu_utama > 0 && subNode.pagu_penunjang > subNode.pagu_utama) {
        const pesanR03 = dapatkanPesanAturan('R-03', 'Struktur Anggaran Tidak Wajar: Porsi Belanja Pendukung Operasional lebih besar daripada Belanja Utama.');
        subNode.subErrors.push(pesanR03);
        totalAnomaliGlobal += 1;
      }

      Object.values(subNode.rekeningRows).forEach(r => {
        if (r.errors && r.errors.length > 0) {
          r.errors.forEach(errTxt => {
            if (!subNode.subErrors.includes(errTxt)) {
              subNode.subErrors.push(errTxt);
            }
          });
        }
      });

      const keySubUnit = subNode.kd_subunit;
      if (!unitHierarkiGroup[keySubUnit]) {
        unitHierarkiGroup[keySubUnit] = {
          kd_subunit: keySubUnit,
          nama_subunit: subNode.nama_subunit,
          subKegiatanList: [],
          total_utama: 0,
          total_penunjang: 0,
          total_pagu: 0
        };
      }
      unitHierarkiGroup[keySubUnit].subKegiatanList.push(subNode);
      unitHierarkiGroup[keySubUnit].total_utama += subNode.pagu_utama;
      unitHierarkiGroup[keySubUnit].total_penunjang += subNode.pagu_penunjang;
      unitHierarkiGroup[keySubUnit].total_pagu += subNode.total_pagu_sub;
    });

    if (Object.keys(unitHierarkiGroup).length === 0) {
      alert("⚠️ Header kolom excel tidak cocok. Gunakan kolom 'Kode SKPD', 'Kode Sub Kegiatan', 'Kode Rekening', 'Anggaran'.");
      setReportGenerated(false);
      return;
    }

    setSubKegiatanTree(treeGroup);
    setSubUnitGroupData(unitHierarkiGroup);
    setTematikReport(tematikGroup);
    setReportGenerated(true);
  };

  const handleSimpanKeDatabase = async () => {
    if (!selectedTahapan || !selectedSkpd || Object.keys(subKegiatanTree).length === 0) {
      alert("Data tidak lengkap untuk disimpan!");
      return;
    }
    setSaving(true);
    try {
      await supabase.from('hsl_verifikasi_rka').delete().eq('kd_skpd_utama', selectedSkpd).eq('id_status_tahapan', selectedTahapan);
      const dataToInsert = Object.values(subKegiatanTree).map((sub) => {
        let tematikKategori = null;
        const angkaMurniSubKeg = sub.kd_sub_kegiatan.replace(/[^0-9]/g, '');
        if (angkaMurniSubKeg.length > 0) {
          const matchTematik = masterTematik.find(t => {
            const m1 = String(t.kdsubgiat || '').replace(/[^0-9]/g, '');
            const m2 = String(t.kode || '').replace(/[^0-9]/g, '');
            return (m1 === angkaMurniSubKeg) || (m2 === angkaMurniSubKeg);
          });
          if (matchTematik && matchTematik.katagori) tematikKategori = String(matchTematik.katagori).toUpperCase().trim();
        }
        return {
          id_status_tahapan: selectedTahapan, 
          kd_skpd_utama: selectedSkpd,       
          kd_subunit: sub.kd_subunit, 
          nama_file: namaFile,
          nama_bidang_urusan: sub.nama_bidang_urusan,
          nama_sub_kegiatan: sub.nama_sub_kegiatan,
          pagu_utama: sub.pagu_utama,
          pagu_penunjang: sub.pagu_penunjang,
          total_pagu_sub: sub.total_pagu_sub,
          catatan_koreksi: sub.subErrors && sub.subErrors.length > 0 ? sub.subErrors.join(' | ') : null,
          b_pegawai: sub.b_pegawai, b_barjas: sub.b_barjas, b_hibah: sub.b_hibah,
          b_modal_alat: sub.b_modal_alat, b_modal_gedung: sub.b_modal_gedung, b_modal_jalan: sub.b_modal_jalan,
          b_modal_aset_tetap: sub.b_modal_aset_tetap, b_modal_aset_lain: sub.b_modal_aset_lain,
          b_btt: sub.b_btt, b_bagi_hasil: sub.b_bagi_hasil, b_bantuan_keu: sub.b_bantuan_keu,
          kategori_tematik: tematikKategori
        };
      });
      const { error } = await supabase.from('hsl_verifikasi_rka').insert(dataToInsert);
      if (error) throw error;
      alert("✓ BERHASIL diarsipkan ke database permanen.");
    } catch (err) {
      alert("Gagal menyimpan data: " + err.message);
    } finally {
      setSaving(false);
    }
  };

  const handleExportToExcel = () => {
    try {
      const wb = XLSX.utils.book_new();
      
      // Sheet 1: Struktur Anggaran
      const sheet1Data = [];
      Object.values(subUnitGroupData).forEach(unit => {
        sheet1Data.push({ 'Kode/Nama': `[UNIT] [${unit.kd_subunit}] ${unit.nama_subunit}`, 'Belanja Utama': unit.total_utama, 'Belanja Pendukung': unit.total_penunjang, 'Total Pagu': unit.total_pagu });
        unit.subKegiatanList.forEach(sub => {
          sheet1Data.push({ 'Kode/Nama': `   ↳ [SUBKEG] [${sub.kd_sub_kegiatan}] ${sub.nama_sub_kegiatan}`, 'Belanja Utama': sub.pagu_utama, 'Belanja Pendukung': sub.pagu_penunjang, 'Total Pagu': sub.total_pagu_sub });
        });
      });
      const ws1 = XLSX.utils.json_to_sheet(sheet1Data);
      XLSX.utils.book_append_sheet(wb, ws1, "Struktur Per Sub-Unit");

      // Sheet 2: Tematik
      const sheet2Data = Object.values(tematikReport);
      const ws2 = XLSX.utils.json_to_sheet(sheet2Data);
      XLSX.utils.book_append_sheet(wb, ws2, "Kluster Tematik");

      // Sheet 3: Jenis Belanja
      const sheet3Data = Object.values(subKegiatanTree).map(item => ({
        'Sub-Unit': item.nama_subunit, 'Sub-Kegiatan': item.nama_sub_kegiatan, 'Pegawai': item.b_pegawai, 'Barjas': item.b_barjas, 'Hibah': item.b_hibah, 'M. Alat': item.b_modal_alat, 'M. Gedung': item.b_modal_gedung, 'M. Jalan': item.b_modal_jalan, 'M. Aset Tetap': item.b_modal_aset_tetap, 'M. Lain': item.b_modal_aset_lain, 'BTT': item.b_btt, 'Transfer': (item.b_bagi_hasil + item.b_bantuan_keu)
      }));
      const ws3 = XLSX.utils.json_to_sheet(sheet3Data);
      XLSX.utils.book_append_sheet(wb, ws3, "Jenis Belanja");

      // Sheet 4: Pelanggaran Regulasi
      const sheet4Data = Object.values(subKegiatanTree).filter(s => s.subErrors.length > 0).map(s => ({
        'Sub-Unit': s.nama_subunit, 'Sub-Kegiatan': s.nama_sub_kegiatan, 'Pelanggaran': s.subErrors.join(' | ')
      }));
      const ws4 = XLSX.utils.json_to_sheet(sheet4Data.length > 0 ? sheet4Data : [{ 'Status': 'Sempurna. Tidak ada pelanggaran.' }]);
      XLSX.utils.book_append_sheet(wb, ws4, "Log Regulasi");

      XLSX.writeFile(wb, `Hasil_Verifikasi_RKA_${selectedSkpd || 'SKPD'}.xlsx`);
    } catch (error) {
      alert("Gagal mengekspor file excel: " + error.message);
    }
  };

  const hitungPersen = (parsial, total) => {
    if (!total || total === 0) return '0%';
    return `${((parsial / total) * 100).toFixed(1)}%`;
  };

  const currentSkpdObj = listSkpd.find(s => s.kd_skpd_utama === selectedSkpd);
  const currentTahapanObj = listTahapan.find(t => String(t.id) === String(selectedTahapan));
  const judulDinamis = currentSkpdObj && currentTahapanObj ? `${currentSkpdObj.display_nama} - TAHAPAN ${currentTahapanObj.status.toUpperCase()}` : '';

  return (

            <button onClick={handleSimpanKeDatabase} disabled={saving} className="flex items-center gap-2 bg-blue-600 hover:bg-blue-500 disabled:bg-blue-800 text-white font-mono font-bold text-xs px-4 py-2.5 rounded-lg shadow-md uppercase tracking-wider transition-all">
